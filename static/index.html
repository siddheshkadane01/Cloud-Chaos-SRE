Quick Start
# 1. Start an easy episode
curl -X POST /reset -H "Content-Type: application/json" \
  -d '{"task_id": "easy"}'

# 2. Check db-proxy logs
curl -X POST /step -H "Content-Type: application/json" \
  -d '{"action_type":"CHECK_LOGS","target_service":"db-proxy","reason":"Investigating errors"}'

# 3. Restart the root-cause service
curl -X POST /step -H "Content-Type: application/json" \
  -d '{"action_type":"RESTART_SERVICE","target_service":"db-proxy","reason":"db-proxy is root cause"}'

# 4. Grade the episode
curl -X POST /grader -H "Content-Type: application/json" -d "$(curl -s /state)"
